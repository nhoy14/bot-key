# Pure-python Logistic Regression classifier for fallback training
class TinyLinearClassifier:
    def __init__(self):
        self.weights = None
        self.bias = 0.0

    def fit(self, X, y):
        if not X:
            return
        n_features = len(X[0])
        self.weights = [0.0] * n_features
        # Simple Gradient Descent
        for _ in range(100):
            for i in range(len(X)):
                xi = X[i]
                yi = y[i]
                # Dot product
                pred = sum(w * x for w, x in zip(self.weights, xi)) + self.bias
                # Sigmoid
                sig = 1.0 / (1.0 + math.exp(-max(min(pred, 20.0), -20.0)))
                # Error gradient
                err = yi - sig
                # Update weights and bias
                for j in range(n_features):
                    self.weights[j] += 0.05 * err * xi[j]
                self.bias += 0.05 * err

    def predict(self, X):
        preds = []
        for xi in X:
            pred = sum(w * x for w, x in zip(self.weights, xi)) + self.bias
            preds.append(1 if pred > 0.0 else 0)
        return preds

# Real AI Training using yfinance and scikit-learn
def train_real(symbol, period, features_to_use, model_type, test_size, trees, interval="1d"):
    # Map XAUUSD to GC=F (Yahoo Finance Gold Futures Ticker)
    symbol_upper = symbol.upper().replace("/", "").replace("-", "")
    if "XAUUSD" in symbol_upper or "GOLD" in symbol_upper:
        symbol = "GC=F"
    # yfinance download
    df = yf.download(symbol, period=period, interval=interval)
    if df.empty:
        raise ValueError(f"No historical data found for symbol {symbol}")
        
    # Flatten MultiIndex columns if present (common in newer yfinance versions)
    if isinstance(df.columns, pd.MultiIndex):
        df.columns = [col[0] for col in df.columns]
        
    df = df.copy()
    # Cast variables to float
    for col in ['Open', 'High', 'Low', 'Close', 'Volume']:
        if col in df.columns:
            df[col] = df[col].astype(float)
            
    # Calculate indicators
    df['sma'] = df['Close'].rolling(window=15).mean()
    df['ema'] = df['Close'].ewm(span=15, adjust=False).mean()
    
    # RSI 14
    delta = df['Close'].diff()
    gain = (delta.where(delta > 0, 0.0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0.0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-10)
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # MACD
    ema12 = df['Close'].ewm(span=12, adjust=False).mean()
    ema26 = df['Close'].ewm(span=26, adjust=False).mean()
    df['macd'] = ema12 - ema26
    df['macd_signal'] = df['macd'].ewm(span=9, adjust=False).mean()
    df['macd_hist'] = df['macd'] - df['macd_signal']
    
    # Bollinger Bands
    df['bb_mid'] = df['Close'].rolling(window=20).mean()
    df['bb_std'] = df['Close'].rolling(window=20).std()
    df['bb_upper'] = df['bb_mid'] + 2 * df['bb_std']
    df['bb_lower'] = df['bb_mid'] - 2 * df['bb_std']
    
    # Label: Buy (1) if Close in 3 days > Close * 1.005, else Sell (0)
    df['target'] = (df['Close'].shift(-3) > df['Close'] * 1.005).astype(int)
    
    df_clean = df.dropna().copy()
    if len(df_clean) < 30:
        raise ValueError("Too few data points after calculating indicators. Try a longer period.")
        
    # Stationarize features
    feature_mapping = {
        'sma': df_clean['Close'] / df_clean['sma'] - 1.0,
        'ema': df_clean['Close'] / df_clean['ema'] - 1.0,
        'rsi': df_clean['rsi'] / 100.0,
        'macd': df_clean['macd_hist'] / df_clean['Close'],
        'bb': (df_clean['Close'] - df_clean['bb_lower']) / (df_clean['bb_upper'] - df_clean['bb_lower'] + 1e-10) - 0.5
    }
    
    active_features = [f for f in features_to_use if f in feature_mapping]
    if not active_features:
        active_features = ['rsi', 'sma']
        
    X = np.column_stack([feature_mapping[f] for f in active_features])
    y = df_clean['target'].values
    
    # Chronological Split
    split_idx = int(len(df_clean) * (1 - test_size))
    if split_idx < 10 or (len(df_clean) - split_idx) < 10:
        split_idx = int(len(df_clean) * 0.8)
        
    X_train, X_test = X[:split_idx], X[split_idx:]
    y_train, y_test = y[:split_idx], y[split_idx:]
    
    # Model
    if model_type == 'rf':
        model = RandomForestClassifier(n_estimators=trees, random_state=42)
    else:
        model = LogisticRegression(random_state=42)
        
    model.fit(X_train, y_train)
    
    # Predict
    train_preds = model.predict(X_train)
    test_preds = model.predict(X_test)
    
    train_acc = float((train_preds == y_train).mean())
    test_acc = float((test_preds == y_test).mean())
    
    # Combine back
    df_train = df_clean.iloc[:split_idx].copy()
    df_test = df_clean.iloc[split_idx:].copy()
    df_train['pred'] = train_preds
    df_test['pred'] = test_preds
    df_all = pd.concat([df_train, df_test])
    
    # Backtest returns
    daily_returns = df_all['Close'].pct_change().fillna(0.0)
    strategy_returns = daily_returns * df_all['pred'].shift(1).fillna(0.0)
    
    market_cum = (1 + daily_returns).cumprod()
    strategy_cum = (1 + strategy_returns).cumprod()
    
    # Trades extraction with SL / TP
    trades = []
    in_pos = False
    entry_p = 0.0
    entry_d = ""
    sl_p = 0.0
    tp_p = 0.0
    
    closes_list = df_all['Close'].values
    highs_list = df_all['High'].values if 'High' in df_all.columns else closes_list
    lows_list = df_all['Low'].values if 'Low' in df_all.columns else closes_list
    preds_list = df_all['pred'].values
    dates_list = [d.strftime('%Y-%m-%d %H:%M') if interval == '1h' else d.strftime('%Y-%m-%d') for d in df_all.index]
    
    k = 0
    while k < len(df_all):
        pred = preds_list[k]
        close = closes_list[k]
        date_str = dates_list[k]
        
        if not in_pos:
            if pred == 1:
                in_pos = True
                entry_p = close
                entry_d = date_str
                sl_p = entry_p * 0.985 # 1.5% SL
                tp_p = entry_p * 1.030 # 3.0% TP
            k += 1
        else:
            high = highs_list[k]
            low = lows_list[k]
            
            if low <= sl_p:
                in_pos = False
                profit = (sl_p - entry_p) / entry_p
                trades.append({
                    'entry_date': entry_d,
                    'exit_date': date_str,
                    'entry_price': float(entry_p),
                    'sl': float(sl_p),
                    'tp': float(tp_p),
                    'exit_price': float(sl_p),
                    'profit_pct': float(profit),
                    'reason': 'Stop Loss 🛑',
                    'status': 'loss'
                })
            elif high >= tp_p:
                in_pos = False
                profit = (tp_p - entry_p) / entry_p
                trades.append({
                    'entry_date': entry_d,
                    'exit_date': date_str,
                    'entry_price': float(entry_p),
                    'sl': float(sl_p),
                    'tp': float(tp_p),
                    'exit_price': float(tp_p),
                    'profit_pct': float(profit),
                    'reason': 'Take Profit 🎯',
                    'status': 'win'
                })
            elif pred == 0:
                in_pos = False
                profit = (close - entry_p) / entry_p
                trades.append({
                    'entry_date': entry_d,
                    'exit_date': date_str,
                    'entry_price': float(entry_p),
                    'sl': float(sl_p),
                    'tp': float(tp_p),
                    'exit_price': float(close),
                    'profit_pct': float(profit),
                    'reason': 'AI Signal 🤖',
                    'status': 'win' if profit > 0 else 'loss'
                })
            k += 1
            
    if in_pos:
        close = closes_list[-1]
        profit = (close - entry_p) / entry_p
        trades.append({
            'entry_date': entry_d,
            'exit_date': dates_list[-1],
            'entry_price': float(entry_p),
            'sl': float(sl_p),
            'tp': float(tp_p),
            'exit_price': float(close),
            'profit_pct': float(profit),
            'reason': 'End of Data ⏳',
            'status': 'win' if profit > 0 else 'loss'
        })
        
    win_t = [t for t in trades if t['profit_pct'] > 0]
    win_rate = len(win_t) / len(trades) if trades else 0.0
    
    std_ret = strategy_returns.std()
    sharpe = float((strategy_returns.mean() / (std_ret + 1e-10)) * math.sqrt(252)) if std_ret > 0 else 0.0
    
    g_profits = sum(t['profit_pct'] for t in trades if t['profit_pct'] > 0)
    g_losses = sum(abs(t['profit_pct']) for t in trades if t['profit_pct'] < 0)
    profit_factor = float(g_profits / (g_losses + 1e-10)) if g_losses > 0 else (999.0 if g_profits > 0 else 1.0)
    
    chart_data = []
    for k in range(len(df_all)):
        chart_data.append({
            'date': dates_list[k],
            'price': float(closes_list[k]),
            'market_return': float(market_cum.iloc[k] - 1.0) * 100.0,
            'strategy_return': float(strategy_cum.iloc[k] - 1.0) * 100.0,
            'signal': int(preds_list[k])
        })
        
    return {
        'success': True,
        'backend': 'real',
        'metrics': {
            'train_acc': train_acc * 100.0,
            'test_acc': test_acc * 100.0,
            'win_rate': win_rate * 100.0,
            'total_trades': len(trades),
            'strategy_return': float(strategy_cum.iloc[-1] - 1.0) * 100.0 if not strategy_cum.empty else 0.0,
            'market_return': float(market_cum.iloc[-1] - 1.0) * 100.0 if not market_cum.empty else 0.0,
            'sharpe': sharpe,
            'profit_factor': profit_factor
        },
        'chart_data': chart_data,
        'trades': trades[-50:]
    }

# Synthetic AI Training fallback
def train_synthetic(symbol, period, features_to_use, model_type, test_size, trees, interval="1d"):
    size_map = {'1mo': 30, '3mo': 90, '6mo': 180, '1y': 365, '2y': 730}
    n_days = size_map.get(period, 365)
    
    symbol_upper = symbol.upper().replace("/", "").replace("-", "")
    is_gold = "XAU" in symbol_upper or "GOLD" in symbol_upper or "GC=F" in symbol_upper
    
    random.seed(42)
    prices = []
    if 'BTC' in symbol_upper:
        curr_price = 60000.0
        volatility = 0.025
    elif 'EUR' in symbol_upper:
        curr_price = 1.1000
        volatility = 0.005
    elif is_gold:
        curr_price = 2350.0
        volatility = 0.009
    else:
        curr_price = 150.0
        volatility = 0.012
        
    drift = 0.0005  # positive drift
    
    dates = []
    if interval == "1h":
        today = datetime.datetime.now()
        for d in range(n_days):
            date_str = (today - datetime.timedelta(hours=n_days - d)).strftime('%Y-%m-%d %H:%M')
            dates.append(date_str)
            change = curr_price * (drift + volatility * random.normalvariate(0.0, 1.0))
            curr_price = max(curr_price + change, 0.01)
            prices.append(curr_price)
    else:
        today = datetime.date.today()
        for d in range(n_days):
            date_str = (today - datetime.timedelta(days=n_days - d)).strftime('%Y-%m-%d')
            dates.append(date_str)
            change = curr_price * (drift + volatility * random.normalvariate(0.0, 1.0))
            curr_price = max(curr_price + change, 0.01)
            prices.append(curr_price)
        
    smas = []
    emas = []
    rsis = []
    macd_hists = []
    bb_uppers = []
    bb_lowers = []
    
    for k in range(n_days):
        if k >= 14:
            smas.append(sum(prices[k-14:k+1]) / 15)
        else:
            smas.append(prices[k])
            
        if k == 0:
            emas.append(prices[0])
        else:
            emas.append(0.125 * prices[k] + 0.875 * emas[-1])
            
        if k >= 14:
            changes = [prices[i] - prices[i-1] for i in range(k-13, k+1)]
            gains = sum(c for c in changes if c > 0)
            losses = sum(-c for c in changes if c < 0)
            rs = gains / (losses + 1e-10)
            rsis.append(100.0 - (100.0 / (1.0 + rs)))
        else:
            rsis.append(50.0 + random.uniform(-10, 10))
            
        macd_val = math.sin(k / 6.0) * (prices[k] * 0.008) + random.uniform(-prices[k]*0.001, prices[k]*0.001)
        macd_hists.append(macd_val)
        
        avg = sum(prices[max(0, k-19):k+1]) / min(k+1, 20)
        bb_uppers.append(avg * 1.025)
        bb_lowers.append(avg * 0.975)
        
    feature_mapping = {
        'sma': [prices[k] / smas[k] - 1.0 for k in range(n_days)],
        'ema': [prices[k] / emas[k] - 1.0 for k in range(n_days)],
        'rsi': [rsis[k] / 100.0 for k in range(n_days)],
        'macd': [macd_hists[k] / prices[k] for k in range(n_days)],
        'bb': [(prices[k] - bb_lowers[k]) / (bb_uppers[k] - bb_lowers[k] + 1e-10) - 0.5 for k in range(n_days)]
    }
    
    targets = []
    for k in range(n_days):
        if k < n_days - 3:
            targets.append(1 if prices[k+3] > prices[k] * 1.004 else 0)
        else:
            targets.append(0)
            
    start_offset = 20
    p_prices = prices[start_offset:]
    p_dates = dates[start_offset:]
    p_targets = targets[start_offset:]
    
    p_features = {}
    for feat in features_to_use:
        if feat in feature_mapping:
            p_features[feat] = feature_mapping[feat][start_offset:]
    if not p_features:
        p_features['rsi'] = feature_mapping['rsi'][start_offset:]
        p_features['sma'] = feature_mapping['sma'][start_offset:]
        
    keys = list(p_features.keys())
    X = []
    for i in range(len(p_prices)):
        X.append([p_features[k][i] for k in keys])
    y = p_targets
    
    split_idx = int(len(X) * (1 - test_size))
    X_train, X_test = X[:split_idx], X[split_idx:]
    y_train, y_test = y[:split_idx], y[split_idx:]
    
    clf = TinyLinearClassifier()
    clf.fit(X_train, y_train)
    
    train_preds = clf.predict(X_train)
    test_preds = clf.predict(X_test)
    
    train_acc = sum(1 for p, t in zip(train_preds, y_train) if p == t) / len(y_train)
    test_acc = sum(1 for p, t in zip(test_preds, y_test) if p == t) / len(y_test)
    
    preds = train_preds + test_preds
    
    daily_returns = [0.0]
    for k in range(1, len(p_prices)):
        daily_returns.append((p_prices[k] - p_prices[k-1]) / p_prices[k-1])
        
    strategy_returns = [0.0]
    for k in range(1, len(p_prices)):
        strategy_returns.append(daily_returns[k] * preds[k-1])
        
    market_cum = [1.0]
    strategy_cum = [1.0]
    for k in range(1, len(p_prices)):
        market_cum.append(market_cum[-1] * (1.0 + daily_returns[k]))
        strategy_cum.append(strategy_cum[-1] * (1.0 + strategy_returns[k]))
        
    trades = []
    in_pos = False
    entry_p = 0.0
    entry_d = ""
    sl_p = 0.0
    tp_p = 0.0
    
    k = 0
    while k < len(p_prices):
        pred = preds[k]
        close = p_prices[k]
        date_str = p_dates[k]
        
        if not in_pos:
            if pred == 1:
                in_pos = True
                entry_p = close
                entry_d = date_str
                sl_p = entry_p * 0.985 # 1.5% SL
                tp_p = entry_p * 1.030 # 3.0% TP
            k += 1
        else:
            if close <= sl_p:
                in_pos = False
                profit = (sl_p - entry_p) / entry_p
                trades.append({
                    'entry_date': entry_d,
                    'exit_date': date_str,
                    'entry_price': float(entry_p),
                    'sl': float(sl_p),
                    'tp': float(tp_p),
                    'exit_price': float(sl_p),
                    'profit_pct': float(profit),
                    'reason': 'Stop Loss 🛑',
                    'status': 'loss'
                })
            elif close >= tp_p:
                in_pos = False
                profit = (tp_p - entry_p) / entry_p
                trades.append({
                    'entry_date': entry_d,
                    'exit_date': date_str,
                    'entry_price': float(entry_p),
                    'sl': float(sl_p),
                    'tp': float(tp_p),
                    'exit_price': float(tp_p),
                    'profit_pct': float(profit),
                    'reason': 'Take Profit 🎯',
                    'status': 'win'
                })
            elif pred == 0:
                in_pos = False
                profit = (close - entry_p) / entry_p
                trades.append({
                    'entry_date': entry_d,
                    'exit_date': date_str,
                    'entry_price': float(entry_p),
                    'sl': float(sl_p),
                    'tp': float(tp_p),
                    'exit_price': float(close),
                    'profit_pct': float(profit),
                    'reason': 'AI Signal 🤖',
                    'status': 'win' if profit > 0 else 'loss'
                })
            k += 1
            
    if in_pos:
        close = p_prices[-1]
        profit = (close - entry_p) / entry_p
        trades.append({
            'entry_date': entry_d,
            'exit_date': p_dates[-1],
            'entry_price': float(entry_p),
            'sl': float(sl_p),
            'tp': float(tp_p),
            'exit_price': float(close),
            'profit_pct': float(profit),
            'reason': 'End of Data ⏳',
            'status': 'win' if profit > 0 else 'loss'
        })
        
    win_t = [t for t in trades if t['profit_pct'] > 0]
    win_rate = len(win_t) / len(trades) if trades else 0.0
    
    mean_ret = sum(strategy_returns) / len(strategy_returns)
    variance = sum((r - mean_ret)**2 for r in strategy_returns) / len(strategy_returns)
    std_ret = math.sqrt(variance)
    sharpe = float((mean_ret / (std_ret + 1e-10)) * math.sqrt(252)) if std_ret > 0 else 0.0
    
    g_profits = sum(t['profit_pct'] for t in trades if t['profit_pct'] > 0)
    g_losses = sum(abs(t['profit_pct']) for t in trades if t['profit_pct'] < 0)
    profit_factor = float(g_profits / (g_losses + 1e-10)) if g_losses > 0 else (999.0 if g_profits > 0 else 1.0)
    
    chart_data = []
    for k in range(len(p_prices)):
        chart_data.append({
            'date': p_dates[k],
            'price': float(p_prices[k]),
            'market_return': float(market_cum[k] - 1.0) * 100.0,
            'strategy_return': float(strategy_cum[k] - 1.0) * 100.0,
            'signal': int(preds[k])
        })
        
    return {
        'success': True,
        'backend': 'synthetic',
        'metrics': {
            'train_acc': train_acc * 100.0,
            'test_acc': test_acc * 100.0,
            'win_rate': win_rate * 100.0,
            'total_trades': len(trades),
            'strategy_return': float(strategy_cum[-1] - 1.0) * 100.0,
            'market_return': float(market_cum[-1] - 1.0) * 100.0,
            'sharpe': sharpe,
            'profit_factor': profit_factor
        },
        'chart_data': chart_data,
        'trades': trades[-50:]
    }

class Handler(BaseHTTPRequestHandler):

    def log_message(self, fmt, *args):
        print(f"  {self.address_string()} → {fmt % args}")

    def do_OPTIONS(self):
        self._send(200)

    def do_GET(self):
        import os
        import urllib.parse
        parsed = urllib.parse.urlparse(self.path)       
        clean_path = parsed.path.strip("/")
        if not clean_path:
            clean_path = "index.html"
            
        # Security check: prevent directory traversal
        safe_path = os.path.basename(clean_path)
        file_path = os.path.join(os.path.dirname(os.path.abspath(__file__)), safe_path)
        
        if os.path.exists(file_path) and os.path.isfile(file_path):
            try:
                # Determine content type
                content_type = "text/plain"
                if file_path.endswith(".html"):
                    content_type = "text/html; charset=utf-8"
                elif file_path.endswith(".png"):
                    content_type = "image/png"
                elif file_path.endswith(".jpg") or file_path.endswith(".jpeg"):
                    content_type = "image/jpeg"
                elif file_path.endswith(".svg"):
                    content_type = "image/svg+xml"
                elif file_path.endswith(".js"):
                    content_type = "application/javascript"
                elif file_path.endswith(".css"):
                    content_type = "text/css"
                elif file_path.endswith(".json"):
                    content_type = "application/json"
                
                with open(file_path, "rb") as f:
                    content = f.read()
                
                self.send_response(200)
                self.send_header("Content-Type", content_type)
                self.send_header("Access-Control-Allow-Origin", "*")
                self.end_headers()
                self.wfile.write(content)
            except Exception as e:
                self.send_response(500)
                self.send_header("Content-Type", "application/json; charset=utf-8")
                self.end_headers()
                self.wfile.write(json.dumps({"error": f"Failed to load file: {str(e)}"}).encode("utf-8"))
        else:
            self._send(404)
            self.wfile.write(json.dumps({"error": "File not found"}).encode("utf-8"))

    def do_POST(self):
        if self.path == "/chat":
            length = int(self.headers.get("Content-Length", 0))
            body   = self.rfile.read(length)

            try:
                data     = json.loads(body)
                messages = data.get("messages", [])
                if not messages:
                    raise ValueError("messages is empty")
            except Exception as e:
                self._send(400)
                self.wfile.write(json.dumps({"error": str(e)}).encode())
                return
        elif self.path == "/train":
            length = int(self.headers.get("Content-Length", 0))
            body   = self.rfile.read(length)
            try:
                data = json.loads(body)
                symbol = data.get("symbol", "BTC-USD")
                period = data.get("period", "1y")
                interval = data.get("interval", "1d")
                features = data.get("features", ["rsi", "sma"])
                model_type = data.get("model_type", "rf")
                test_size = float(data.get("test_size", 0.2))
                trees = int(data.get("trees", 100))
                
                if ML_LIBS_AVAILABLE:
                    try:
                        res = train_real(symbol, period, features, model_type, test_size, trees, interval)
                    except Exception as e:
                        print(f"  ⚠️ Real training failed ({e}), falling back to synthetic.")
                        res = train_synthetic(symbol, period, features, model_type, test_size, trees, interval)
                        res['fallback_reason'] = str(e)
                else:
                    res = train_synthetic(symbol, period, features, model_type, test_size, trees, interval)
                
                self._send(200)
                self.wfile.write(json.dumps(res, ensure_ascii=False).encode("utf-8"))
                return
            except Exception as e:
                print(f"  ❌ Train Error: {e}")
                self._send(500)
                self.wfile.write(json.dumps({"error": str(e)}).encode())
                return
        else:
            self._send(404)
            self.wfile.write(b'{"error":"not found"}')
            return

        # Convert chat history to Gemini format
        # Gemini uses "user" / "model" roles (not "assistant")
        gemini_contents = []
        for m in messages:
            role = "model" if m["role"] == "assistant" else "user"
            gemini_contents.append({
                "role": role,
                "parts": [{"text": m["content"]}]
            })

        payload = json.dumps({
            "system_instruction": {
                "parts": [{"text": SYSTEM}]
            },
            "contents": gemini_contents,
            "generationConfig": {
                "maxOutputTokens": 1000,
                "temperature": 0.7,
            }
        }).encode("utf-8")

        # Rebuild URL with current key (in case key was set after import)
        url = (
            f"https://generativelanguage.googleapis.com/v1beta/models/"
            f"{MODEL}:generateContent?key={API_KEY}"
        )

        req = urllib.request.Request(
            url,
            data    = payload,
            headers = {"Content-Type": "application/json"},
            method  = "POST",
        )

        try:
            with urllib.request.urlopen(req, timeout=30) as resp:
                result = json.loads(resp.read().decode("utf-8"))

            # Extract text from Gemini response
            reply = (
                result
                .get("candidates", [{}])[0]
                .get("content", {})
                .get("parts", [{}])[0]
                .get("text", "សូមទោស មានបញ្ហា។")
            )

            self._send(200)
            self.wfile.write(json.dumps({
                "reply": reply
            }, ensure_ascii=False).encode("utf-8"))

        except urllib.error.HTTPError as e:
            err = e.read().decode()
            print(f"  ❌ Gemini error {e.code}: {err}")
            self._send(502)
            self.wfile.write(json.dumps({"error": f"Gemini {e.code}: {err}"}).encode())

        except Exception as e:
            print(f"  ❌ Error: {e}")
            self._send(500)
            self.wfile.write(json.dumps({"error": str(e)}).encode())

    def _send(self, code: int):
        self.send_response(code)
        self.send_header("Content-Type",                "application/json; charset=utf-8")
        self.send_header("Access-Control-Allow-Origin",  "*")
        self.send_header("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
        self.send_header("Access-Control-Allow-Headers", "Content-Type")
        self.end_headers()
